## DATABASE SCHEMA

```sql
CREATE TABLE safety_incidents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id),
  project_id UUID REFERENCES projects(id),

  -- Incident Info
  incident_number VARCHAR(50) UNIQUE,
  occurred_at TIMESTAMPTZ NOT NULL,
  location TEXT,

  -- Severity
  severity TEXT NOT NULL, -- 'near_miss', 'first_aid', 'recordable', 'lost_time', 'fatality'
  incident_type TEXT, -- 'fall', 'struck_by', 'caught_between', 'electrical', etc.

  -- Person
  employee_name VARCHAR(255),
  employee_id UUID REFERENCES auth.users(id),
  employee_age INT,
  ppe_worn BOOLEAN,

  -- Description
  description TEXT NOT NULL,
  root_cause TEXT,
  corrective_actions TEXT,

  -- OSHA Reporting
  is_osha_recordable BOOLEAN DEFAULT false,
  days_away_from_work INT DEFAULT 0,
  reported_to_osha BOOLEAN DEFAULT false,
  reported_at TIMESTAMPTZ,

  -- Documentation
  photos TEXT[],
  witness_statements TEXT[],

  -- Follow-up
  follow_up_required BOOLEAN DEFAULT true,
  follow_up_completed BOOLEAN DEFAULT false,
  follow_up_date DATE,

  reported_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE safety_briefings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL REFERENCES projects(id),
  company_id UUID NOT NULL REFERENCES companies(id),

  -- Briefing Info
  briefing_date DATE NOT NULL,
  conducted_by UUID NOT NULL REFERENCES auth.users(id),
  duration_minutes INT,

  -- Content
  topics_covered TEXT[],
  hazards_identified TEXT[],
  ppe_required TEXT[],
  weather_conditions JSONB,

  -- Attendance
  attendees JSONB, -- [{user_id, name, signature, timestamp}]
  attendance_count INT,

  -- Documentation
  photos TEXT[],
  toolbox_talk_topic TEXT,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE certifications (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id),

  -- Certification Info
  certification_type TEXT NOT NULL, -- 'license', 'insurance', 'osha', 'equipment'
  name VARCHAR(255) NOT NULL,
  number VARCHAR(100),

  -- Holder
  holder_type TEXT NOT NULL, -- 'company', 'employee', 'subcontractor'
  holder_id UUID, -- user_id or subcontractor_id

  -- Validity
  issue_date DATE,
  expiration_date DATE,
  is_active BOOLEAN DEFAULT true,

  -- Alerts
  reminder_days_before INT DEFAULT 60,
  reminder_sent BOOLEAN DEFAULT false,

  -- Documentation
  certificate_url TEXT,

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Auto-mark certifications as inactive when expired
CREATE OR REPLACE FUNCTION check_cert_expiration()
RETURNS void AS $$
BEGIN
  UPDATE certifications
  SET is_active = false
  WHERE expiration_date < CURRENT_DATE
  AND is_active = true;
END;
$$ LANGUAGE plpgsql;

-- Run daily via cron
```
